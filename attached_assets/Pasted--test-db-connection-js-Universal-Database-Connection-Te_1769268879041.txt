// test-db-connection.js - Universal Database Connection Tester
const fs = require('fs');
require('dotenv').config();

console.log('🗄️  DATABASE CONNECTION TESTER');
console.log('================================\n');

// Detect database type from environment variables
function detectDatabaseType() {
  if (process.env.MONGODB_URI || process.env.MONGO_URL) return 'mongodb';
  if (process.env.DATABASE_URL && process.env.DATABASE_URL.includes('postgres')) return 'postgresql';
  if (process.env.DATABASE_URL && process.env.DATABASE_URL.includes('mysql')) return 'mysql';
  if (process.env.REDIS_URL) return 'redis';
  if (process.env.DATABASE_URL && process.env.DATABASE_URL.includes('sqlite')) return 'sqlite';
  return 'unknown';
}

// MongoDB Connection Test
async function testMongoDB() {
  try {
    const mongoose = require('mongoose');
    const uri = process.env.MONGODB_URI || process.env.MONGO_URL;
    
    console.log('📦 Testing MongoDB Connection...');
    console.log(`URI: ${uri.replace(/:[^:@]+@/, ':****@')}`); // Hide password
    
    await mongoose.connect(uri, {
      useNewUrlParser: true,
      useUnifiedTopology: true,
      serverSelectionTimeoutMS: 5000
    });
    
    console.log('✅ MongoDB Connected Successfully!');
    console.log(`   Database: ${mongoose.connection.db.databaseName}`);
    console.log(`   Host: ${mongoose.connection.host}`);
    console.log(`   Port: ${mongoose.connection.port}`);
    
    // Test write operation
    const collections = await mongoose.connection.db.listCollections().toArray();
    console.log(`   Collections: ${collections.length}`);
    
    // Ping test
    const pingResult = await mongoose.connection.db.admin().ping();
    console.log(`   Ping: ${pingResult.ok === 1 ? '✅ OK' : '❌ Failed'}`);
    
    await mongoose.connection.close();
    return true;
  } catch (error) {
    console.log('❌ MongoDB Connection Failed!');
    console.error(`   Error: ${error.message}`);
    return false;
  }
}

// PostgreSQL Connection Test
async function testPostgreSQL() {
  try {
    const { Client } = require('pg');
    const client = new Client({
      connectionString: process.env.DATABASE_URL,
      ssl: process.env.NODE_ENV === 'production' ? { rejectUnauthorized: false } : false
    });
    
    console.log('🐘 Testing PostgreSQL Connection...');
    
    await client.connect();
    console.log('✅ PostgreSQL Connected Successfully!');
    
    // Get database info
    const dbInfo = await client.query('SELECT version(), current_database(), current_user');
    console.log(`   Database: ${dbInfo.rows[0].current_database}`);
    console.log(`   User: ${dbInfo.rows[0].current_user}`);
    
    // Test query
    const result = await client.query('SELECT NOW() as current_time');
    console.log(`   Server Time: ${result.rows[0].current_time}`);
    
    // List tables
    const tables = await client.query(`
      SELECT table_name 
      FROM information_schema.tables 
      WHERE table_schema = 'public'
    `);
    console.log(`   Tables: ${tables.rows.length}`);
    
    await client.end();
    return true;
  } catch (error) {
    console.log('❌ PostgreSQL Connection Failed!');
    console.error(`   Error: ${error.message}`);
    return false;
  }
}

// MySQL Connection Test
async function testMySQL() {
  try {
    const mysql = require('mysql2/promise');
    const connection = await mysql.createConnection(process.env.DATABASE_URL || {
      host: process.env.DB_HOST,
      user: process.env.DB_USER,
      password: process.env.DB_PASSWORD,
      database: process.env.DB_NAME
    });
    
    console.log('🐬 Testing MySQL Connection...');
    
    console.log('✅ MySQL Connected Successfully!');
    
    // Get database info
    const [rows] = await connection.query('SELECT DATABASE() as db, USER() as user, VERSION() as version');
    console.log(`   Database: ${rows[0].db}`);
    console.log(`   User: ${rows[0].user}`);
    console.log(`   Version: ${rows[0].version}`);
    
    // List tables
    const [tables] = await connection.query('SHOW TABLES');
    console.log(`   Tables: ${tables.length}`);
    
    await connection.end();
    return true;
  } catch (error) {
    console.log('❌ MySQL Connection Failed!');
    console.error(`   Error: ${error.message}`);
    return false;
  }
}

// SQLite Connection Test
async function testSQLite() {
  try {
    const sqlite3 = require('sqlite3').verbose();
    const dbPath = process.env.SQLITE_PATH || './database.sqlite';
    
    console.log('💾 Testing SQLite Connection...');
    console.log(`   Path: ${dbPath}`);
    
    return new Promise((resolve) => {
      const db = new sqlite3.Database(dbPath, (err) => {
        if (err) {
          console.log('❌ SQLite Connection Failed!');
          console.error(`   Error: ${err.message}`);
          resolve(false);
        } else {
          console.log('✅ SQLite Connected Successfully!');
          
          // Check if database file exists
          const exists = fs.existsSync(dbPath);
          console.log(`   File exists: ${exists ? '✅ Yes' : '❌ No'}`);
          
          // Get table list
          db.all("SELECT name FROM sqlite_master WHERE type='table'", [], (err, tables) => {
            if (!err) {
              console.log(`   Tables: ${tables.length}`);
            }
            db.close();
            resolve(true);
          });
        }
      });
    });
  } catch (error) {
    console.log('❌ SQLite Connection Failed!');
    console.error(`   Error: ${error.message}`);
    return false;
  }
}

// Redis Connection Test
async function testRedis() {
  try {
    const redis = require('redis');
    const client = redis.createClient({
      url: process.env.REDIS_URL
    });
    
    console.log('🔴 Testing Redis Connection...');
    
    await client.connect();
    console.log('✅ Redis Connected Successfully!');
    
    // Ping test
    const pong = await client.ping();
    console.log(`   Ping: ${pong === 'PONG' ? '✅ OK' : '❌ Failed'}`);
    
    // Get Redis info
    const info = await client.info('server');
    const version = info.match(/redis_version:([^\r\n]+)/)?.[1];
    console.log(`   Version: ${version}`);
    
    // Test read/write
    await client.set('test_key', 'test_value');
    const value = await client.get('test_key');
    console.log(`   Read/Write Test: ${value === 'test_value' ? '✅ OK' : '❌ Failed'}`);
    await client.del('test_key');
    
    await client.quit();
    return true;
  } catch (error) {
    console.log('❌ Redis Connection Failed!');
    console.error(`   Error: ${error.message}`);
    return false;
  }
}

// Main execution
async function runDatabaseTests() {
  const dbType = detectDatabaseType();
  
  console.log(`Detected Database Type: ${dbType.toUpperCase()}\n`);
  
  let success = false;
  
  switch(dbType) {
    case 'mongodb':
      success = await testMongoDB();
      break;
    case 'postgresql':
      success = await testPostgreSQL();
      break;
    case 'mysql':
      success = await testMySQL();
      break;
    case 'sqlite':
      success = await testSQLite();
      break;
    case 'redis':
      success = await testRedis();
      break;
    default:
      console.log('❌ Unknown database type or missing environment variables');
      console.log('\nExpected environment variables:');
      console.log('  MongoDB: MONGODB_URI or MONGO_URL');
      console.log('  PostgreSQL: DATABASE_URL (postgres://...)');
      console.log('  MySQL: DATABASE_URL (mysql://...)');
      console.log('  SQLite: SQLITE_PATH');
      console.log('  Redis: REDIS_URL');
  }
  
  console.log('\n================================');
  console.log(success ? '✅ ALL TESTS PASSED' : '❌ TESTS FAILED');
  process.exit(success ? 0 : 1);
}

// Run tests
runDatabaseTests();
